\hypersetup{
    colorlinks,
    linkcolor={red!50!black},
    citecolor={blue!80!black},
    urlcolor={blue!80!black}

% Arduino-Farben definieren
\definecolor{arduino_keyword}{RGB}{0, 90, 164}    % Blau für #define, if, etc.
\definecolor{arduino_function}{RGB}{211, 84, 0}   % Orange/Braun für setup, loop, pinMode
\definecolor{arduino_literal}{RGB}{0, 151, 156}   % Teal für HIGH, LOW, INPUT
\definecolor{arduino_comment}{RGB}{149, 165, 166} % Grau für Kommentare
\definecolor{arduino_string}{RGB}{0, 135, 142}    % Dunkelgrün für Strings
\definecolor{arduino_bg}{RGB}{248, 249, 249}      % Ganz helles Grau für den Hintergrund

% Code-Style konfigurieren
\lstdefinestyle{myArduino}{
    language=C++, % Arduino basiert auf C++
    backgroundcolor=\color{arduino_bg},
    commentstyle=\color{arduino_comment}\itshape,
    keywordstyle=\color{arduino_keyword}\bfseries,
    numberstyle=\tiny\color{arduino_comment},
    stringstyle=\color{arduino_string},
    basicstyle=\ttfamily\small,
    breakatwhitespace=false,         
    breaklines=true,                 
    captionpos=b,                    
    keepspaces=true,                 
    numbers=left,                    
    numbersep=8pt,                  
    showspaces=false,                
    showstringspaces=false,
    showtabs=false,                  
    tabsize=2,
    % Eigene Arduino-Schlüsselwörter hinzufügen, die C++ nicht kennt
    morekeywords=[2]{INPUT, OUTPUT, HIGH, LOW, INPUT_PULLUP, FALLING, RISING, CHANGE},
    keywordstyle=[2]\color{arduino_literal},
    morekeywords=[3]{setup, loop, pinMode, digitalWrite, digitalRead, attachInterrupt, digitalPinToInterrupt, delay},
    keywordstyle=[3]\color{arduino_function}
}
